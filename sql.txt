DROP PROCEDURE IF EXISTS `deleteorder`;
CREATE DEFINER=`root`@`localhost` PROCEDURE `deleteorder`(
	IN `parorderid` int,
	IN `parmodifiedby` int
)
LANGUAGE SQL
NOT DETERMINISTIC
CONTAINS SQL
SQL SECURITY DEFINER
COMMENT ''
BEGIN 
	CREATE TEMPORARY TABLE temppayment
	SELECT OM.orderid,SM.shiftid,sum(PTM.CreditAmount) AS creditamount,SUM((case when PTM.paymodeid=1 then ifnull(PTM.creditamount,0.00) ELSE 0 END)) AS cashamount
	from ordermaster OM
	JOIN paymentmaster PM ON OM.orderid=PM.orderid AND OM.isdeleted=0 AND PM.isdeleted=0
	JOIN paymenttransactionmaster PTM ON PM.paymentid=PTM.paymentid AND PTM.isdeleted=0
	JOIN shiftmaster SM ON OM.shiftid=SM.shiftid
	WHERE OM.orderid=parorderid
	ORDER BY OM.orderid,SM.shiftid;
	
	UPDATE shiftmaster sm
	JOIN temppayment tp ON sm.shiftid=tp.shiftid
	SET SM.totalpaymentamount=(ifnull(SM.totalpaymentamount,0.00)-IFNULL(tp.creditamount,0.00)),
	SM.closingbalance=(ifnull(SM.closingbalance,0.00)-IFNULL(tp.cashamount,0.00));
	
	DROP TABLE temppayment;
	
	update customermaster cm
	JOIN ordermaster OM ON cm.customerid=OM.customerid
    join paymentmaster pm on pm.orderid=om.orderid and pm.isdeleted=0
    join paymenttransactionmaster ptm on ptm.paymentid=ptm.paymentid and ptm.isdeleted=0 and ptm.paymodeid=9
	SET cm.outstandingamt=(IFNULL(cm.outstandingamt,0.00)-IFNULL(om.grandtotal,0.00))
	WHERE OM.orderid=parorderid and om.isdeleted=0;
    
	INSERT INTO customeraccountdetails (customerid, creditamount, balance, paymentdate,datekey,createdby,createddate,description,locationid,paymodeid,companyid) SELECT om.customerid,grandtotal,outstandingamt,now(),date_format(now(),'%Y%m%d'),parmodifiedby,now(),concat(billno,' is cancelled'),om.locationid,0,om.companyid FROM ordermaster om 
	join paymentmaster pm on pm.orderid=om.orderid and pm.isdeleted=0 and om.orderid=parorderid and om.isdeleted=0
	join paymenttransactionmaster ptm on ptm.paymentid=pm.paymentid and ptm.isdeleted=0 and ptm.paymodeid=9
	join customermaster cm on cm.customerid=om.customerid and cm.isdeleted=0;
	
    
	update ordermaster OM
	JOIN orderproductdetails OPD ON OM.orderid=OPD.orderid
	join orderproducttaxdetails OPT ON OPD.orderproductid=OPT.orderproductid
	join paymentmaster PM ON OM.orderid=PM.orderid
	join paymenttransactionmaster PTM ON PM.paymentid=PTM.paymentid
	join ticketmaster TM ON OPD.ticketid=TM.ticketid
	SET OM.isdeleted=1,OM.issync=0,OM.modifiedby=parmodifiedby,OM.modifieddate=now(),
	OPD.isdeleted=1,OPD.modifiedby=parmodifiedby,OPD.modifieddate=now(),
	OPT.isdeleted=1,OPT.modifiedby=parmodifiedby,OPT.modifieddate=now(),
	PM.isdeleted=1,PM.modifiedby=parmodifiedby,PM.modifieddate=now(),
	PTM.isdeleted=1,PTM.modifiedby=parmodifiedby,PTM.modifieddate=now(),
	TM.isdeleted=1,TM.modifiedby=parmodifiedby,TM.modifieddate=now()
	WHERE OM.orderid=parorderid;
	
	update ordermaster OM
	JOIN orderproductdetails OPD ON OM.orderid=OPD.orderid
	join orderproductmodifiers OPM ON OPD.orderproductid=OPM.orderproductdetailsid
	SET OPM.isdeleted=1,OPM.modifiedby=parmodifiedby,OPM.modifieddate=now()
	WHERE OM.orderid=parorderid;
END;
